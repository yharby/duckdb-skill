# H3 DuckDB Extension

H3 discrete global grid system — index points and geometries to hexagonal cells.

> Extension: `h3` (FROM community). H3 v4.4.1, DuckDB v1.5.0. Apache 2.0.

## Setup

```sql
INSTALL h3 FROM community;
LOAD h3;
```

## Input Types

H3 cell IDs accepted as `UBIGINT` (preferred, fastest), `BIGINT`, or `VARCHAR`.

```sql
-- All equivalent:
SELECT h3_cell_to_latlng(586265647244115967);           -- UBIGINT
SELECT h3_cell_to_latlng(CAST(586265647244115967 AS BIGINT));  -- BIGINT
SELECT h3_cell_to_latlng('822d57fffffffff');             -- VARCHAR
```

## Coordinate Conversion

```sql
-- Point → H3 cell
SELECT h3_latlng_to_cell(40.7128, -74.0060, 9) AS cell_id;  -- (lat, lng, resolution)
SELECT h3_latlng_to_cell_string(40.7128, -74.0060, 9);       -- returns VARCHAR

-- H3 cell → point
SELECT h3_cell_to_lat(cell_id), h3_cell_to_lng(cell_id);
SELECT h3_cell_to_latlng(cell_id);  -- returns STRUCT(lat, lng)

-- String ↔ integer conversion
SELECT h3_string_to_h3('822d57fffffffff');   -- VARCHAR → UBIGINT
SELECT h3_h3_to_string(586265647244115967);  -- UBIGINT → VARCHAR
```

## Cell Hierarchy

```sql
-- Parent / children
SELECT h3_cell_to_parent(cell_id, 5);           -- coarser resolution
SELECT h3_cell_to_children(cell_id, 11);         -- finer resolution (returns LIST)
SELECT h3_cell_to_children_size(cell_id, 11);    -- count without materializing
SELECT h3_cell_to_center_child(cell_id, 11);     -- center child only

-- Child position (index within parent)
SELECT h3_cell_to_child_pos(cell_id, 5);         -- position in parent at res 5
SELECT h3_child_pos_to_cell(pos, parent, 11);    -- reconstruct from position
```

## Cell Properties

```sql
SELECT h3_get_resolution(cell_id);
SELECT h3_get_base_cell_number(cell_id);
SELECT h3_is_valid_cell(cell_id);       -- BOOLEAN
SELECT h3_is_pentagon(cell_id);          -- 12 pentagons exist per resolution
SELECT h3_is_res_class_iii(cell_id);
SELECT h3_get_icosahedron_faces(cell_id);  -- LIST of face indices
```

## Grid Operations (Neighbors, Rings, Paths)

```sql
-- k-ring (disk): all cells within k steps
SELECT h3_grid_disk(cell_id, 2);              -- LIST of cells within 2 steps
SELECT h3_grid_disk_distances(cell_id, 2);    -- LIST of STRUCT(cell, distance)
SELECT h3_grid_disk_unsafe(cell_id, 2);       -- faster, may fail near pentagons

-- Hollow ring at exactly k distance
SELECT h3_grid_ring(cell_id, 2);
SELECT h3_grid_ring_unsafe(cell_id, 2);

-- Path between two cells
SELECT h3_grid_path_cells(cell_a, cell_b);    -- LIST of cells along path

-- Distance (grid steps, not meters)
SELECT h3_grid_distance(cell_a, cell_b);

-- Max disk size (for pre-allocation)
SELECT h3_max_grid_disk_size(k);
```

## Geometry Output

```sql
-- Cell boundary as WKT/WKB (for use with spatial extension)
SELECT h3_cell_to_boundary_wkt(cell_id);
SELECT h3_cell_to_boundary_wkb(cell_id);

-- Multiple cells → MultiPolygon
SELECT h3_cells_to_multi_polygon_wkt(cell_list);
SELECT h3_cells_to_multi_polygon_wkb(cell_list);

-- Edge boundary
SELECT h3_directed_edge_to_boundary_wkt(edge_id);
SELECT h3_directed_edge_to_boundary_wkb(edge_id);
```

## Polygon → Cells (Polyfill)

```sql
-- Fill a polygon with H3 cells at resolution 9
SELECT h3_polygon_wkt_to_cells('POLYGON((-74 40, -73 40, -73 41, -74 41, -74 40))', 9);
SELECT h3_polygon_wkb_to_cells(geom_blob, 9);

-- String variants (return VARCHAR cell IDs)
SELECT h3_polygon_wkt_to_cells_string(wkt, 9);
SELECT h3_polygon_wkb_to_cells_string(wkb, 9);

-- Experimental (newer algorithm)
SELECT h3_polygon_wkt_to_cells_experimental(wkt, 9);
SELECT h3_polygon_wkb_to_cells_experimental(wkb, 9);
```

## Compaction

```sql
-- Compact a set of cells (replace complete groups with parent)
SELECT h3_compact_cells(cell_list);

-- Expand back to target resolution
SELECT h3_uncompact_cells(compacted_list, 9);
```

## Directed Edges

```sql
-- Get all edges from a cell
SELECT h3_origin_to_directed_edges(cell_id);

-- Edge endpoints
SELECT h3_get_directed_edge_origin(edge_id);
SELECT h3_get_directed_edge_destination(edge_id);
SELECT h3_directed_edge_to_cells(edge_id);  -- STRUCT(origin, destination)

-- Create edge between neighbors
SELECT h3_cells_to_directed_edge(cell_a, cell_b);

-- Check adjacency
SELECT h3_are_neighbor_cells(cell_a, cell_b);
SELECT h3_is_valid_directed_edge(edge_id);
```

## Metrics

```sql
-- Cell area (in km² or m²)
SELECT h3_cell_area(cell_id, 'km^2');    -- exact area of specific cell
SELECT h3_cell_area(cell_id, 'm^2');

-- Edge length
SELECT h3_edge_length(edge_id, 'km');
SELECT h3_edge_length(edge_id, 'm');

-- Great circle distance between two points
SELECT h3_great_circle_distance(lat1, lng1, lat2, lng2, 'km');

-- Average values per resolution
SELECT h3_get_hexagon_area_avg(9, 'km^2');         -- ~0.1053 km²
SELECT h3_get_hexagon_edge_length_avg(9, 'km');     -- ~0.174 km

-- Total cells at a resolution
SELECT h3_get_num_cells(9);  -- 4,842,432,842
```

## Common Patterns

```sql
-- Index points to H3, aggregate, and visualize
SELECT h3_latlng_to_cell(lat, lon, 7) AS cell,
       count() AS cnt,
       avg(value) AS avg_val
FROM points
GROUP BY ALL;

-- Spatial join via H3 (fast alternative to ST_Intersects)
SELECT a.*, b.*
FROM table_a a
JOIN table_b b ON h3_latlng_to_cell(a.lat, a.lon, 9) = h3_latlng_to_cell(b.lat, b.lon, 9);

-- Expand to geometry for visualization
SELECT h3_cell_to_boundary_wkt(cell) AS geom, cnt
FROM aggregated_cells;

-- K-ring smoothing
SELECT unnest(h3_grid_disk(cell_id, 1)) AS neighbor, value
FROM cells;
```
